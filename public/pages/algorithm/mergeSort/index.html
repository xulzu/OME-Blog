# 归并排序

// 归并排序是一种时间复杂度o n*logn 空间复杂度o n的一种排序算法
//具体的原理为将排序的对象不停的一分为二，当分到最小单元时合并两个部分使得较小单元先排列好顺序

class mergeSort{
    constructor(arr=[]){
        this.OrignArr=arr
        this.sorted=this.mainMergSort(arr)

    }
    // 合并入口
    mainMergSort(arr){
        return this.splitTwoPart(arr,0,arr.length-1)
    }
    /**
    * @desc 合并两个有序的部分
    * @param {} 
    * @return {}
    **/
    mergeSortedPart(larr,rarr){
        // l-m为左半部分，m-r为右半部分
        let curAns=[]
        while(larr.length&&rarr.length){
            if(larr[0]<=rarr[0]){
                curAns.push(larr.shift())
            }else{
                curAns.push(rarr.shift())
            }
        }
        curAns.push(...[...larr,...rarr])
        return curAns
    }
    /**
    * @desc 把数组分割成两部分
    * @param {} 
    * @return {}
    **/
    splitTwoPart(arr){
        if(arr.length<=1) return arr
        let l=0,r=arr.length-1
        let mid=(l+r)>>1
        return this.mergeSortedPart(this.splitTwoPart(arr.slice(l,mid+1)),this.splitTwoPart(arr.slice(mid+1)))
    }
}
